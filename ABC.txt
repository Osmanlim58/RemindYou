<!DOCTYPE html>
<html lang="de">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>RemindYou - Fristen im Fokus</title>
    <script src="https://cdn.tailwindcss.com"></script>
</head>
<body class="bg-gray-100">
    <div id="root" class="p-4 max-w-4xl mx-auto"></div>
    <script>
        let user = null;
        let assets = JSON.parse(localStorage.getItem('assets') || '[]');
        
        // Demo-Admin
        if (assets.length === 0) {
            assets = [
                { id: 1, name: "Gabelstapler B12", status: "orange", date: "2026-04-15", pruefart: "TÜV", pruefer: "Klaus" },
                { id: 2, name: "Feuerlöscher", status: "red", date: "2025-03-01", pruefart: "Brandschutz", pruefer: "Max" }
            ];
        }

        function save() { localStorage.setItem('assets', JSON.stringify(assets)); }
        
        function render() {
            const html = `
                <h1 class="text-3xl font-bold text-green-600">💡✓ RemindYou</h1>
                <p class="text-gray-500 mb-4">Deine Fristen im Fokus</p>
                
                <div class="grid grid-cols-3 gap-3 mb-4">
                    <div class="bg-red-100 p-3 rounded text-center"><b>${assets.filter(a=>a.status==='red').length}</b><br>Überfällig</div>
                    <div class="bg-orange-100 p-3 rounded text-center"><b>${assets.filter(a=>a.status==='orange').length}</b><br>Bald fällig</div>
                    <div class="bg-green-100 p-3 rounded text-center"><b>${assets.filter(a=>a.status==='green').length}</b><br>In Ordnung</div>
                </div>
                
                <button onclick="newAsset()" class="bg-green-600 text-white px-4 py-2 rounded mb-3">+ Neues Gerät</button>
                
                <div class="bg-white rounded shadow">
                    <table class="w-full">
                        <tr class="bg-gray-200"><th class="p-2">Gerät</th><th>Status</th><th>Prüfart</th><th>Prüfer</th></tr>
                        ${assets.map(a => `<tr class="border-t"><td class="p-2">${a.name}</td><td class="${a.status==='red'?'text-red-600':a.status==='orange'?'text-orange-600':'text-green-600'}">${a.status}</td><td>${a.pruefart}</td><td>${a.pruefer}</td></tr>`).join('')}
                    </table>
                </div>
                <div class="text-center text-sm text-gray-400 mt-4">Demo: cguler23@gmail.com / Remindly2026!</div>
            `;
            document.getElementById('root').innerHTML = html;
        }
        
        window.newAsset = () => {
            let name = prompt("Geräte Name:");
            if(name) {
                assets.push({ id: Date.now(), name, status: "green", date: "2026-12-31", pruefart: "Neu", pruefer: "Admin" });
                save();
                render();
            }
        };
        
        render();
    </script>
</body>
</html>
